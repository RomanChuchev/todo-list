Example
const [[d, h, m, s], setTime] = useState([days, hours, min, sec]);
const tick = () => {
  if (d === 0 && h === 0 && m === 0 && s === 0) {
    completeTodo();
  } else if (h === 0 && m === 0 && s === 0) {
    setTime([d - 1, 23, 59, 59]);
  } else if (m === 0 && s === 0) {
    setTime([d, h - 1, 59, 59]);
  } else if (s === 0) {
    setTime([d, h, m - 1, 59]);
  } else {
    setTime([d, h, m, s - 1]);
  }
};
